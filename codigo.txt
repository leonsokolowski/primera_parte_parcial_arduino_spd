#define A 9
#define B 10
#define C 4
#define D 5
#define E 6
#define F 8
#define G 7
#define pulsador_mas 13
#define pulsador_menos 12
#define pulsador_reset 11
#define DisplayOn 25
#define decena A4
#define unidad A5
#define Apagados 1

int countDigit = 0;
int sumaPrevia = 1;
int restaPrevia = 1;
int resetPrevia = 1;

void setup() 
{
    pinMode(A, OUTPUT);
    pinMode(B, OUTPUT);
    pinMode(C, OUTPUT);
    pinMode(D, OUTPUT);
    pinMode(E, OUTPUT);
    pinMode(F, OUTPUT);
    pinMode(G, OUTPUT);

    pinMode(pulsador_mas, INPUT_PULLUP);
    pinMode(pulsador_menos, INPUT_PULLUP);
    pinMode(pulsador_reset, INPUT_PULLUP);

    pinMode(decena, OUTPUT);
    pinMode(unidad, OUTPUT);
    digitalWrite(decena, LOW);
    digitalWrite(unidad, LOW);

    Serial.begin(9600);
}

void loop()
{
    
    int pressed = KeyPressed();
    if (pressed == pulsador_mas)
    {
        countDigit++;
        if (countDigit > 99)
        {
            countDigit = 0;
        }
    }
    else if (pressed == pulsador_menos)
    {
        countDigit--;
        if (countDigit < 0)
        {
            countDigit = 99;
        }
    }
    else if (pressed == pulsador_reset)
    {
        countDigit = 0;
    }
    Serial.println(countDigit);
    printCount(countDigit);
  delay(10);
}

void printDigit(int digit)
{
  //Recibe el numero que se deberia mostrar en pantalla
  //Apaga todos los Led's, y prende los necesarios para c/NÂ°
  //No devuelve nada
    digitalWrite(A, LOW);
    digitalWrite(B, LOW);
    digitalWrite(C, LOW);
    digitalWrite(D, LOW);
    digitalWrite(E, LOW);
    digitalWrite(F, LOW);
    digitalWrite(G, LOW);

    switch (digit)
    {
    case 0:
        digitalWrite(A, HIGH);
        digitalWrite(B, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(D, HIGH);
        digitalWrite(E, HIGH);
        digitalWrite(F, HIGH);
        break;
    case 1:
        digitalWrite(B, HIGH);
        digitalWrite(C, HIGH);
        break;
    case 2:
        digitalWrite(A, HIGH);
        digitalWrite(B, HIGH);
        digitalWrite(G, HIGH);
        digitalWrite(D, HIGH);
        digitalWrite(E, HIGH);
        break;
    case 3:
        digitalWrite(A, HIGH);
        digitalWrite(B, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(G, HIGH);
        digitalWrite(D, HIGH);
        break;
    case 4:
        digitalWrite(B, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(G, HIGH);
        digitalWrite(F, HIGH);
        break;
    case 5:
        digitalWrite(A, HIGH);
        digitalWrite(F, HIGH);
        digitalWrite(G, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(D, HIGH);
        break;
    case 6:
        digitalWrite(A, HIGH);
        digitalWrite(G, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(D, HIGH);
        digitalWrite(E, HIGH);
        digitalWrite(F, HIGH);
        break;
    case 7:
        digitalWrite(A, HIGH);
        digitalWrite(B, HIGH);
        digitalWrite(C, HIGH);
        break;
    case 8:
        digitalWrite(A, HIGH);
        digitalWrite(B, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(D, HIGH);
        digitalWrite(E, HIGH);
        digitalWrite(F, HIGH);
        digitalWrite(G, HIGH);
        break;
    case 9:
        digitalWrite(A, HIGH);
        digitalWrite(G, HIGH);
        digitalWrite(C, HIGH);
        digitalWrite(B, HIGH);
        digitalWrite(F, HIGH);
        break;
    }
}

void prendeDigito(int digito)
{
  //Recibe el numero que se deberia mostrar en pantalla
  //Apaga los display, y prende con el numero en su valor posicional
  //No devuelve nada
    digitalWrite(unidad, HIGH);
    digitalWrite(decena, HIGH);

    if (digito == unidad)
    {
        digitalWrite(unidad, HIGH);
      	digitalWrite(decena, LOW);
      delay(85);
    }
    else if (digito == decena)
    {
        digitalWrite(decena, HIGH);
      	digitalWrite(unidad, LOW);
     delay(85);
    }
  	
}

void printCount(int count)
{
  //Recibe un numero
  //Lo transforma a unidad o decena
  //No devuelve nada
    int DECENA = count / 10;
    int UNIDAD = count % 10;

    prendeDigito(decena);
    printDigit(DECENA);
  	
    prendeDigito(unidad);
    printDigit(UNIDAD);
}

int KeyPressed(void)
{
  //No recibe nada
  //Evalua cual boton fue pulsado
  //Devuelve una respuesta dependiendo el boton
    int suma = digitalRead(pulsador_mas);
    int resta = digitalRead(pulsador_menos);
    int reset = digitalRead(pulsador_reset);
    if (suma)
    {
        sumaPrevia = 1;
    }
    if (resta)
    {
        restaPrevia = 1;
    }
    if (reset)
    {
        resetPrevia = 1;
    }

    if (suma == 0 && suma != sumaPrevia)
    {
        sumaPrevia = suma;
        return pulsador_mas;
    }
    if (resta == 0 && resta != restaPrevia)
    {
        restaPrevia = resta;
        return pulsador_menos;
    }
    if (reset == 0 && reset != resetPrevia)
    {
        resetPrevia = reset;
        return pulsador_reset;
    }
    return 0;
}
